- detached_house_bias
- townhouse_bias
- apartment_bias
- condominium_bias
- rental_bias

- worker_housing_bias
- elite_housing_bias
- redevelopment_bias


# ============================================================
# 3. HELIOS
# ============================================================

housing_posture:
- inner elite ring housing preserved
- general ring apartment_and_townhouse_mixed
- outer worker_housing_high
- redevelopment_bias_high in old poor/slave outer belt


# ============================================================
# 4. NOVA
# ============================================================

housing_posture:
- high apartment and condominium bias
- executive luxury housing concentrated south of ASIC
- strong rental and metro-support housing
- asic employee residential concentration
- mixed_use moderate in outer commercial belts


# ============================================================
# 5. SEIWA
# ============================================================

housing_posture:
- balanced detached/townhouse/apartment mix
- rail-oriented residential clustering
- family housing high
- student housing high in education cities
- frontier east retains future expansion capacity


# ============================================================
# 6. GLADIA
# ============================================================

housing_posture:
- controlled residential support housing
- soldier and family housing high
- low luxury housing share
- function-first residential structure
- military-adjacent worker housing high


# ============================================================
# 7. ORPHEUS
# ============================================================

housing_posture:
- coastal housing high
- medium-density island housing
- mixed residential + craft + fishery support
- tourism-adjacent housing moderate
- ritual/coastal settlement clustering


# ============================================================
# 8. AURELIA
# ============================================================

housing_posture:
- current runtime degraded
- controlled believer residential legacy
- fragmented post-collapse housing
- low stable redevelopment completion


# ============================================================
# 9. FREE CITIES UNION
# ============================================================

housing_posture:
- mixed_use_housing_high
- residential/commercial blend high
- small workshop adjacency common
- detached share lower than suburban monarchic states
- rental and civic mixed housing common


# ============================================================
# 10. FINAL RULE
# ============================================================

Initial housing seed must reflect social structure,
zoning posture, and city function.
