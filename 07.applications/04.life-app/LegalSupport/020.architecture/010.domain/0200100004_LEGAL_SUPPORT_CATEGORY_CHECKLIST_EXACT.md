category_checklists:

  inheritance:
    checklist_items:
      - 被相続人情報を整理したか

      - 相談したいことを整理したか

      - 保有書類を整理したか

  consumer:
    checklist_items:

  debt:
    checklist_items:

      - 次回対応を整理したか

  labor:
    checklist_items:
      - 勤務先を整理したか
      - 雇用形態を整理したか

      - 相談したいことを整理したか

  housing:
